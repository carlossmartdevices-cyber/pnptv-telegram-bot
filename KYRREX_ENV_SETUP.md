# Kyrrex Payment Integration - Environment Variables

# Add these to your .env file for Kyrrex cryptocurrency payments

# Kyrrex API Configuration
KYRREX_API_URL=https://api.kyrrex.com/v1
KYRREX_API_KEY=your_kyrrex_api_key_here
KYRREX_SECRET=your_kyrrex_secret_key_here
KYRREX_WALLET_ADDRESS=your_kyrrex_wallet_address_here

# These are already in your existing .env - just make sure they exist
BOT_URL=https://your-domain.com
